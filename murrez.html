-- Oyuncunun karakterini al local player = game.Players.LocalPlayer local character = player.Character or player.CharacterAdded:Wait() -- Karakterin mevcut konumunu al local position = character.PrimaryPart.Position -- Başlangıç ve hedef pozisyonları local baslangicPozisyon = position local hedefPozisyon = Vector3.new(700.0066528320312, 3.5371994972229004, 166.16146850585938) -- Hedefe giderken kullanılacak adım sayısı ve süresi local adimSayisi = 50 -- Toplam adım sayısı local hareketSuresi = 2 -- Toplam hareket süresi (saniye) -- Adım boyutu ve süresi local adimBoyutu = (hedefPozisyon - baslangicPozisyon) / adimSayisi local adimSuresi = hareketSuresi / adimSayisi -- Karakteri hedefe doğru adım adım taşıyan fonksiyon local function HedefeGit() local primaryPart = character.PrimaryPart if not primaryPart then warn("Karakterin PrimaryPart özelliği bulunamadı!") return end local simdikiPozisyon = primaryPart.Position local yeniPozisyon = simdikiPozisyon + adimBoyutu -- Karakterin pozisyonunu güncelle primaryPart.CFrame = CFrame.new(yeniPozisyon) -- Hala hedefe ulaşmadıysak bir sonraki adım için bekleyelim if (hedefPozisyon - yeniPozisyon).magnitude > adimBoyutu.magnitude then wait(adimSuresi) HedefeGit() else -- Hedefe ulaşıldığında son bir güncelleme yapalım primaryPart.CFrame = CFrame.new(hedefPozisyon) print("Hedefe başarıyla ulaşıldı!") end end -- Hedefe git fonksiyonunu çağır HedefeGit()
